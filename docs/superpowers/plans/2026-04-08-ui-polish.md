# UI Polish Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Elevate ExcelChat to a professional data-tool aesthetic (Retool/Metabase reference) with a unified design system, premium chart styling, consistent animations, and zero emojis throughout.

**Architecture:** A single `ui_theme.py` module exports color tokens, a CSS string, and a shared Plotly layout dict. Every page imports from it. Graph node/edge rendering is upgraded in `graph_engine.py`. Page files are renamed to remove emoji prefixes. All inline CSS in `app.py` is replaced with the unified system.

**Tech Stack:** Streamlit CSS injection via `st.markdown(unsafe_allow_html=True)`, Plotly `graph_objects` + `plotly.express`, Python dataclasses for design tokens, `re` for parsing LLM insight card markdown.

---

| File | Action | Responsibility |
|---|---|---|
| `ui_theme.py` | CREATE | Design tokens, CSS string, `PLOTLY_LAYOUT` dict, helper functions |

def kpi_card(value: str, label: str, warning: bool = False) -> str:
    """Return HTML for a single KPI metric card."""
    cls = "kpi-card warning" if warning else "kpi-card"
    return (
        f'<div class="{cls}">'
        f'<div class="kpi-value">{value}</div>'
        f'<div class="kpi-label">{label}</div>'
        f'</div>'
    )


def section_header(title: str) -> str:
    """Return HTML for a section header with bottom border rule."""
    return f'<div class="section-header">{title}</div>'


def insight_card_html(title: str, confidence: str, evidence: str, forecast: str) -> str:
    """Return HTML for a prediction insight card."""
    badge_cls = f"badge badge-{confidence.lower()}" if confidence.lower() in ("high", "medium", "low") else "badge badge-low"
    return (
        f'<div class="insight-card prediction">'
        f'<div style="display:flex;justify-content:space-between;align-items:flex-start;margin-bottom:0.5rem;">'
        f'<div class="insight-card-title">{title}</div>'
        f'<span class="{badge_cls}">{confidence}</span>'
        f'</div>'
        f'<div class="insight-field-label">Evidence</div>'
        f'<div class="insight-field-value">{evidence}</div>'
        f'<div class="insight-field-label">Forecast</div>'
        f'<div class="insight-field-value">{forecast}</div>'
        f'</div>'
    )


def cause_card_html(title: str, metric: str, driver: str, signal: str) -> str:
    """Return HTML for a causes analysis card."""
    return (
        f'<div class="insight-card cause">'
        f'<div class="insight-card-title">{title}</div>'
        f'<div class="insight-field-label">Affected Metric</div>'
        f'<div class="insight-field-value">{metric}</div>'
        f'<div class="insight-field-label">Likely Driver</div>'
        f'<div class="insight-field-value">{driver}</div>'
        f'<div class="insight-field-label">Supporting Signal</div>'
        f'<div class="insight-field-value">{signal}</div>'
        f'</div>'
    )

Add this function to the page (before rendering logic):

```python
def _parse_prediction_cards(text: str) -> list[dict]:
    """
    Parse structured LLM markdown into card dicts.
    Expects blocks like:
        ### Prediction N: Title
        **Confidence:** High
        **Evidence:** ...
        **Forecast:** ...
    Returns list of dicts with keys: title, confidence, evidence, forecast
    """
    cards = []
    blocks = re.split(r'###\s+Prediction\s+\d+[:\.]?\s*', text, flags=re.IGNORECASE)
    for block in blocks[1:]:  # skip text before first heading
        lines = block.strip().splitlines()
        title = lines[0].strip() if lines else "Insight"
        confidence = "Medium"
        evidence = ""
        forecast = ""
        current_field = None
        for line in lines[1:]:
            line = line.strip()
            if re.match(r'\*\*Confidence[:\*]*', line, re.IGNORECASE):
                confidence = re.sub(r'\*\*Confidence[:\*]*\s*', '', line, flags=re.IGNORECASE).strip(" *")
            elif re.match(r'\*\*Evidence[:\*]*', line, re.IGNORECASE):
                current_field = "evidence"
                evidence = re.sub(r'\*\*Evidence[:\*]*\s*', '', line, flags=re.IGNORECASE).strip(" *")
            elif re.match(r'\*\*Forecast[:\*]*', line, re.IGNORECASE):
                current_field = "forecast"
                forecast = re.sub(r'\*\*Forecast[:\*]*\s*', '', line, flags=re.IGNORECASE).strip(" *")
            elif line and current_field == "evidence":
                evidence += " " + line
            elif line and current_field == "forecast":
                forecast += " " + line
        if title:
            cards.append(dict(title=title, confidence=confidence,
                              evidence=evidence.strip(), forecast=forecast.strip()))
    return cards


def _parse_cause_cards(text: str) -> list[dict]:
    """
    Parse cause analysis LLM markdown into card dicts.
    Expects blocks like:
        ### Factor N: Title
        **Affected metric:** col
        **Likely driver:** ...
        **Supporting signal:** ...
    Returns list of dicts with keys: title, metric, driver, signal
    """
    cards = []
    blocks = re.split(r'###\s+Factor\s+\d+[:\.]?\s*', text, flags=re.IGNORECASE)
    for block in blocks[1:]:
        lines = block.strip().splitlines()
        title = lines[0].strip() if lines else "Factor"
        metric = driver = signal = ""
        current_field = None
        for line in lines[1:]:
            line = line.strip()
            if re.match(r'\*\*Affected metric[:\*]*', line, re.IGNORECASE):
                current_field = "metric"
                metric = re.sub(r'\*\*Affected metric[:\*]*\s*', '', line, flags=re.IGNORECASE).strip(" *")
            elif re.match(r'\*\*Likely driver[:\*]*', line, re.IGNORECASE):
                current_field = "driver"
                driver = re.sub(r'\*\*Likely driver[:\*]*\s*', '', line, flags=re.IGNORECASE).strip(" *")
            elif re.match(r'\*\*Supporting signal[:\*]*', line, re.IGNORECASE):
                current_field = "signal"
                signal = re.sub(r'\*\*Supporting signal[:\*]*\s*', '', line, flags=re.IGNORECASE).strip(" *")
            elif line and current_field == "driver":
                driver += " " + line
            elif line and current_field == "signal":
                signal += " " + line
        if title:
            cards.append(dict(title=title, metric=metric.strip(),
                              driver=driver.strip(), signal=signal.strip()))
    return cards
```
